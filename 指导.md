```python
def _create_input_group(self, parent, label, default="", placeholder="", required=False):
    """创建美化的输入组"""
    group = ctk.CTkFrame(parent, fg_color="transparent")
    group.pack(fill="x", pady=5)
    
    # 标签
    label_frame = ctk.CTkFrame(group, fg_color="transparent")
    label_frame.pack(fill="x")
    
    label_text = f"{label} " + ("*" if required else "")
    lbl = ctk.CTkLabel(
        label_frame,
        text=label_text,
        font=ctk.CTkFont(size=12, weight="normal"),
        text_color=("gray40", "gray70")
    )
    lbl.pack(side="left")
    
    if required:
        req_label = ctk.CTkLabel(
            label_frame,
            text="(必填)",
            font=ctk.CTkFont(size=10),
            text_color="red"
        )
        req_label.pack(side="left", padx=(5, 0))
    
    # 输入框
    entry = ctk.CTkEntry(
        group,
        placeholder_text=placeholder,
        height=36,
        corner_radius=8,
        border_width=1,
        fg_color=("gray97", "gray20"),
        border_color=("gray70", "gray50"),
        text_color=("gray10", "gray90")
    )
    entry.pack(fill="x", pady=(5, 0))
    entry.insert(0, default)
    
    return entry
```

def _create_card_frame(self, parent, title, icon=""):
    """创建卡片式的分组框架"""
    card = ctk.CTkFrame(

def _create_input_group(self, parent, label, default="", placeholder="", required=False):
    """创建美化的输入组"""
    group = ctk.CTkFrame(parent, fg_color="transparent")
    group.pack(fill="x", pady=4)
    
    label_frame = ctk.CTkFrame(group, fg_color="transparent")
    label_frame.pack(fill="x")
    
    label_text = f"{label} {'*' if required else ''}"
    ctk.CTkLabel(
        label_frame,
        text=label_text,
        font=ctk.CTkFont(size=12),
        text_color=("gray40", "gray70")
    ).pack(side="left")
    
    entry = ctk.CTkEntry(
        group,
        placeholder_text=placeholder,
        height=36,
        corner_radius=8,
        border_width=1,
        fg_color=("gray97", "gray20"),
        border_color=("gray70", "gray50")
    )
    entry.pack(fill="x", pady=(4, 0))
    entry.insert(0, default)
    
    return entry

def _create_inline_input(self, parent, label, placeholder, default, row, col):
    """创建行内输入框"""
    frame = ctk.CTkFrame(parent, fg_color="transparent")
    frame.grid(row=row, column=col, sticky="ew", padx=(0 if col == 0 else 10, 0 if col == 1 else 10), pady=5)
    frame.grid_columnconfigure(0, weight=1)
    
    if label:
        ctk.CTkLabel(
            frame,
            text=label,
            font=ctk.CTkFont(size=12),
            text_color=("gray40", "gray70")
        ).pack(anchor="w", pady=(0, 2))
    
    entry = ctk.CTkEntry(
        frame,
        placeholder_text=placeholder,
        height=34,
        corner_radius=8,
        border_width=1,
        fg_color=("gray97", "gray20"),
        border_color=("gray70", "gray50")
    )
    entry.pack(fill="x")
    entry.insert(0, str(default))
    
    return entry

def _create_switch(self, parent, text, variable, row, col):
    """创建开关选项"""
    switch_frame = ctk.CTkFrame(parent, fg_color="transparent")
    switch_frame.grid(row=row, column=col, sticky="w", padx=(0, 20), pady=5)
    
    switch = ctk.CTkSwitch(
        switch_frame,
        text=text,
        variable=variable,
        font=ctk.CTkFont(size=13),
        progress_color="#2B87E3"
    )
    switch.pack(side="left")
    
    return switch

## 主要改进点

1. **卡片式布局**：使用圆角卡片分组，增加视觉层次
2. **网格布局**：参数设置使用2列网格，节省空间
3. **图标点缀**：使用 emoji 图标增加可读性
4. **色彩系统**：统一的色彩方案（蓝色主题）
5. **间距优化**：调整各元素间距，更加透气
6. **字体层次**：标题、正文使用不同的字重和大小
7. **输入框美化**：添加占位符、边框和圆角
8. **状态指示**：状态标签使用图标前缀

这些改进保持了 `customtkinter` 的原生特性，同时大幅提升了界面的现代感和专业度。
